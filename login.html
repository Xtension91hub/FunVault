import pygame import random import sys # Initialize Pygame pygame.init() # Screen dimensions WIDTH = 400 HEIGHT = 600 SCREEN = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Flappy Bird") # Colors WHITE = (255, 255, 255) GREEN = (0, 255, 0) BLUE = (0, 0, 255) # Load images BIRD = pygame.image.load("bird.png") # Use a bird image here BG = pygame.image.load("background.png") # Background image PIPE_TOP = pygame.image.load("pipe_top.png") # Top pipe image PIPE_BOTTOM = pygame.image.load("pipe_bottom.png") # Bottom pipe image # Resize images BIRD = pygame.transform.scale(BIRD, (50, 35)) BG = pygame.transform.scale(BG, (WIDTH, HEIGHT)) PIPE_TOP = pygame.transform.scale(PIPE_TOP, (80, 400)) PIPE_BOTTOM = pygame.transform.scale(PIPE_BOTTOM, (80, 400)) # Bird position bird_x = 60 bird_y = 300 bird_y_velocity = 0 g = 0.5 # Pipe data pipe_gap = 150 pipe_x = WIDTH pipe_height = random.randint(100, 400) # Score score = 0 font = pygame.font.SysFont(None, 36) # Clock clock = pygame.time.Clock() def draw_pipes(pipe_x, pipe_height): SCREEN.blit(PIPE_TOP, (pipe_x, pipe_height - 400)) SCREEN.blit(PIPE_BOTTOM, (pipe_x, pipe_height + pipe_gap)) def main(): global bird_y, bird_y_velocity, pipe_x, pipe_height, score running = True while running: clock.tick(60) for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: bird_y_velocity = -8 # Flap the bird # Bird movement bird_y_velocity += g bird_y += bird_y_velocity # Move pipes pipe_x -= 4 if pipe_x < -80: pipe_x = WIDTH pipe_height = random.randint(100, 400) score += 1 # Collision check if bird_y > HEIGHT or bird_y < 0 or ( pipe_x < bird_x + 50 < pipe_x + 80 and (bird_y < pipe_height or bird_y > pipe_height + pipe_gap)): print("Game Over") pygame.quit() sys.exit() # Draw everything SCREEN.blit(BG, (0, 0)) draw_pipes(pipe_x, pipe_height) SCREEN.blit(BIRD, (bird_x, bird_y)) # Display score score_text = font.render(f"Score: {score}", True, WHITE) SCREEN.blit(score_text, (10, 10)) pygame.display.update() if __name__ == "__main__": main()
